git init					(Initialize git repository)
git status					(View files to add in git staging area)
git add <file name>				(Add file in git staging area)
git commit -m "initial commit"			(commit file form staging area to git repository)
git log						(check commits)
git checkout <branch name>			(got to a particular branch in git repository)
git branch <branch name>			(make a branch in git repository)
git merge <branch name>				(merge branches)
git remote -v 					(check git remote)
git remote add <name/origin> <github url>	(add git with github repo)
git branch -M master
git push -u <name/origin> main			(push to github)
git pull <name/origin> main			(pull commits from github)
